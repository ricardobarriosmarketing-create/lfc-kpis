@st.cache_data
def load_sales_data():
    """Intenta cargar el CSV real saltando las líneas con errores de formato"""
    try:
        # AQUI ESTÁ LA SOLUCIÓN: on_bad_lines='skip'
        df = pd.read_csv("ADMIN DASHBOARD.xlsx - 📞 Llamadas de venta.csv", on_bad_lines='skip')
        df['Fecha Llamada '] = pd.to_datetime(df['Fecha Llamada '], errors='coerce')
        df['Total Vendido'] = pd.to_numeric(df['Total Vendido'], errors='coerce').fillna(0)
        df = df[df['Fecha Llamada '].dt.month.isin([1, 2, 3, 4])]
        return df
    except FileNotFoundError:
        np.random.seed(42)
        fechas = pd.date_range(start="2026-01-01", end="2026-04-30")
        datos = []
        for fecha in fechas:
            num_llamadas = np.random.randint(5, 15)
            for _ in range(num_llamadas):
                asistio = np.random.choice(['ASISTIO', 'NO ASISTIO'], p=[0.4, 0.6])
                status = np.random.choice(['CERRADO', 'ABIERTO', 'PERDIDO']) if asistio == 'ASISTIO' else 'ABIERTO'
                vendido = np.random.choice([0, 4000, 5000]) if status == 'CERRADO' else 0
                datos.append({'Fecha Llamada ': fecha, 'Outcome': asistio, 'Status': status, 'Total Vendido': vendido, 'Closer': np.random.choice(['Daniel', 'Luis', 'Ashley', 'Daisy'])})
        return pd.DataFrame(datos)
